    <h1>欢迎来到我的网页！</h1>
    <button id="startButton">입장</button>

    <div class="calendar" id="calendar" style="display: none;">
        <h2>小日历</h2>
        <div class="dateDisplay" id="currentDate"></div>
    </div>

    <div class="notes" id="notes" style="display: none;">
        <h2>便签板</h2>
        <textarea id="noteInput" placeholder="在这里写下你的小事情..."></textarea>
        <button id="submitNote">发送</button>
        <div id="noteList"></div>
    </div>

    <script>
        document.getElementById('startButton').onclick = function() {
            document.getElementById('calendar').style.display = 'block';
            document.getElementById('notes').style.display = 'block';
            this.style.display = 'none'; // 隐藏按钮
            displayDate();
            loadNotes();
        };

        function displayDate() {
            const date = new Date();
            const options = { year: 'numeric', month: 'long', day: 'numeric' };
            document.getElementById('currentDate').innerText = date.toLocaleDateString('zh-CN', options);
        }

        document.getElementById('submitNote').onclick = function() {
            const noteInput = document.getElementById('noteInput');
            const note = noteInput.value;
            if (note) {
                const timestamp = new Date().toLocaleString();
                const noteEntry = `${timestamp}: ${note}`;
                saveNote(noteEntry);
                noteInput.value = ''; // 清空输入框
                loadNotes(); // 更新便签列表
            }
        };

        function saveNote(note) {
            const notes = JSON.parse(localStorage.getItem('notes')) || [];
            notes.push(note);
            localStorage.setItem('notes', JSON.stringify(notes));
        }

        function loadNotes() {
            const notes = JSON.parse(localStorage.getItem('notes')) || [];
            document.getElementById('noteList').innerHTML = notes.map(n => `<div>${n}</div>`).join('');
        }
    </script>
